SELECT gadget_name
FROM gadgets
WHERE gadget_name LIKE '%iPhone%' AND category_name = 'Mobile';

SELECT gadget_name
FROM gadgets
WHERE gadget_name LIKE 'ASUS%' AND category_name = 'Computer';
